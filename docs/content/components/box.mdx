## Import

```tsx code
import { Box } from '@marigold/components';
```

## Live-Code Example

```tsx live
<Box bg="success" p="8px" width="25%">
  <Text>Box Example</Text>
</Box>
```

## Usage

```tsx
<Box>I am basic box!</Box>
```

```tsx
<Box as="button" bg="primary">
  I am a button box!
</Box>
```

```tsx
<Box as="p" p="16px">
  I am a paragraph with padding!
</Box>
```
